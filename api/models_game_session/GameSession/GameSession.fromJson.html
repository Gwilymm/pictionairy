factory GameSession.fromJson(Map<String, dynamic> json) {
  return GameSession(
    id: json['id'].toString(),
    gameStarterId: json['gameStarterId'],
    redTeamPlayers: List<String>.from(json['redTeamPlayers'] ?? []),
    blueTeamPlayers: List<String>.from(json['blueTeamPlayers'] ?? []),
  );
}
